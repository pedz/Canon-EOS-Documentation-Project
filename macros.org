#+NAME: menu
#+HEADER: :var item="Detect priority AE while AF"
#+BEGIN_SRC elisp :exports none :var table=menu-table
  (let* ((len (length table))
         (index 0)
         (str (replace-regexp-in-string "\n *" " " item))
         tab
         temp)
    (while (and (< index len) ; >
                (not (string= (elt (elt table index) 3) str)))
      (setq index (1+ index)))
    (if (= index len)
        (error "String '%s' not found" str))
    (setq temp (elt table index))
    (setq tab (replace-regexp-in-string "^[Tt]ab \\([0-9]+\\)/[0-9]+ .*$" "\\1" (elt temp 2)))
    (format "~%s~ menu item found in tab %s of the %s menu"
            (elt temp 3)
            tab
            (elt temp 1)))
#+END_SRC
